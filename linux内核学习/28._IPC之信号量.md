```
	struct ipc_params sem_params;

static int newary(struct ipc_namespace *ns, struct ipc_params *params)
{
	int retval;
	struct sem_array *sma;
	key_t key = params->key;
	int nsems = params->u.nsems;
	int semflg = params->flg;
	int i;
......
    // 1. 创建strcut sem_array
	sma = sem_alloc(nsems);
......
	sma->sem_perm.mode = (semflg & S_IRWXUGO);
	sma->sem_perm.key = key;
	sma->sem_perm.security = NULL;
......
    // 2. sem_array核心是一个struct sem的数组sems
    // struct sem里包含：
	//int	semval;		当前信号量的数值
	//int	sempid;     最近一个修改信号量的线程PID
	//struct list_head pending_alter;  针对本信号量的修改队列（struct sem_queue）
	//struct list_head pending_const;
	for (i = 0; i < nsems; i++) {
		INIT_LIST_HEAD(&sma->sems[i].pending_alter);
		INIT_LIST_HEAD(&sma->sems[i].pending_const);
		spin_lock_init(&sma->sems[i].lock);
	}
	sma->complex_count = 0;
	sma->use_global_lock = USE_GLOBAL_LOCK_HYSTERESIS;
	INIT_LIST_HEAD(&sma->pending_alter);
	INIT_LIST_HEAD(&sma->pending_const);
	INIT_LIST_HEAD(&sma->list_id);
	sma->sem_nsems = nsems;
	sma->sem_ctime = get_seconds();
	
	// 3. 挂到基树上
	retval = ipc_addid(&sem_ids(ns), &sma->sem_perm, ns->sc_semmni);
......
	ns->used_sems += nsems;
......
	return sma->sem_perm.id;
}
```

## 修改（P/V操作）

```
// 对信号量的一个操作
struct sembuf  
{  
  unsigned short int sem_num;   /* 信号量的序号从0~nsems-1 */  
  short int sem_op;            /* 对信号量的操作，>0, 0, <0 */  
  short int sem_flg;            /* 操作标识：0， IPC_WAIT, SEM_UNDO */ 
};  

// PV操作调用的都是semtimedop:
SYSCALL_DEFINE4(semtimedop, int, semid, struct sembuf __user *, tsops,
		unsigned, nsops, const struct timespec __user *, timeout)
{
	int error = -EINVAL;
	struct sem_array *sma;
	struct sembuf fast_sops[SEMOPM_FAST];
	struct sembuf *sops = fast_sops, *sop;
	struct sem_undo *un;
	int max, locknum;
	bool undos = false, alter = false, dupsop = false;
	struct sem_queue queue;
	unsigned long dup = 0, jiffies_left = 0;
	struct ipc_namespace *ns;

	ns = current->nsproxy->ipc_ns;
......
    // 从用户态将操作拷贝到内核
	if (copy_from_user(sops, tsops, nsops * sizeof(*tsops))) {
		error =  -EFAULT;
		goto out_free;
	}

    // P操作可能被阻塞，如果有超时时间，那么转换成时钟滴答
	if (timeout) {
		struct timespec _timeout;
		if (copy_from_user(&_timeout, timeout, sizeof(*timeout))) {
		}
		jiffies_left = timespec_to_jiffies(&_timeout);
	}
......
	/* On success, find_alloc_undo takes the rcu_read_lock */
	un = find_alloc_undo(ns, semid);
......
    // 找到被操作的sem_array*
	sma = sem_obtain_object_check(ns, semid);
......
    // struct sem_queue queue 表示当前的操作
	queue.sops = sops;  //关联sembuf
	queue.nsops = nsops;
	queue.undo = un;
	queue.pid = task_tgid_vnr(current);
	queue.alter = alter;
	queue.dupsop = dupsop;

    // 实施信号量操作
	error = perform_atomic_semop(sma, &queue);
	
	if (error == 0) { /* non-blocking succesfull path */
	    // 成功了  不用等待
		DEFINE_WAKE_Q(wake_q);
......
        //看本次信号量的变化可以激活等待队列中的哪些sem_queue
		do_smart_update(sma, sops, nsops, 1, &wake_q);
......
        //唤醒这些等待的进程
		wake_up_q(&wake_q);
		goto out_free;
	}
	/*
	 * We need to sleep on this operation, so we put the current
	 * task into the pending queue and go to sleep.
	 */
	// 不成功 需要等待
	if (nsops == 1) {
		struct sem *curr;
		curr = &sma->sems[sops->sem_num];
......
        // 加到本信号量的pending
		list_add_tail(&queue.list,
						&curr->pending_alter);
......
	} else {
......
        // 对整个信号量集合的pending 
		list_add_tail(&queue.list, &sma->pending_alter);
......
	}

    // 通过do while来等待
	do {
		queue.status = -EINTR;
		queue.sleeper = current;
    
		__set_current_state(TASK_INTERRUPTIBLE); //可以被信号中断(可以被kill)
		if (timeout)
			jiffies_left = schedule_timeout(jiffies_left); //让出CPU
		else
			schedule(); //让出CPU
......
		/*
		 * If an interrupt occurred we have to clean up the queue.
		 */
		 // 等待超时 返回-1
		if (timeout && jiffies_left == 0)
			error = -EAGAIN;
	} while (error == -EINTR && !signal_pending(current)); /* spurious */
......
}
```

## SEM_UNDO

防止一个占用信号量资源的进程异常退出，导致其它等待的进程永远等不到信号量释放的问题。
